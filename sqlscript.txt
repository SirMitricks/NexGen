CREATE TABLE IF NOT EXISTS reasons (
    id serial PRIMARY KEY,
    title varchar(255) NOT NULL,
    description text NOT NULL,
    learn_more_url varchar(255) DEFAULT '/Home'
);

create table if not exists ourServices (
	id serial primary key,
	img_url varchar(255) DEFAULT '#',
	title varchar(255) not null,
	description text not null,
	book_a_call varchar(255) DEFAULT '/Home',
	price varchar(255) not null
);

create table if not exists ourWorks (
	id serial primary key,
	img_title varchar(255) DEFAULT '#',
	title varchar(255) not null,
	details varchar(255) DEFAULT '/Home',
	category varchar(255) not null,
	time_taken varchar(255) not null,
	description text not null,
	img_url varchar(255) DEFAULT '#',
	technologies varchar(255) not null,
	book_a_call varchar(255) DEFAULT '/Home'
);

create table if not exists testimonials (
    id serial primary key,
    title varchar(255) not null,
    description text not null,
    ava_img varchar(255) DEFAULT '#',
    name varchar(255) not null,
    prof varchar(255) not null,
    learn_more_url varchar(255) DEFAULT '/Home'
);

CREATE TABLE IF NOT EXISTS Design (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255),
    category VARCHAR(100),
    read_time VARCHAR(20),
    author VARCHAR(100),
    description TEXT,
    published_date DATE,
    has_read_full_button BOOLEAN DEFAULT TRUE
);

ALTER TABLE public.design ADD COLUMN IF NOT EXISTS image varchar NULL;

CREATE TABLE IF NOT EXISTS Business (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255),
    category VARCHAR(100),
    read_time VARCHAR(20),
    author VARCHAR(100),
    description TEXT,
    published_date DATE,
    has_read_full_button BOOLEAN DEFAULT TRUE
);

ALTER TABLE public.business ADD COLUMN IF NOT EXISTS image varchar NULL;

CREATE TABLE IF NOT EXISTS Development (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255),
    category VARCHAR(100),
    read_time VARCHAR(20),
    author VARCHAR(100),
    description TEXT,
    published_date DATE,
    has_read_full_button BOOLEAN DEFAULT TRUE
);

ALTER TABLE public.development ADD COLUMN IF NOT EXISTS image varchar NULL;

CREATE TABLE IF NOT EXISTS faq (
    id SERIAL PRIMARY KEY,
    question TEXT NOT NULL,
    answer TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);
